1.
---일반적으로 집계함수는 GROUP BY 절과 같이 사용되지만 아래와 같이 테이블 전체가 
---하나의 그룹이 되는 경우에는 GROUP BY 절 없이 단독으로도 사용 가능하다. 
SELECT COUNT(*) AS 전체행수, COUNT(HEIGHT)AS 키건수,
MAX(HEIGHT)AS 최대키, MIN(HEIGHT)AS최소키, ROUND(AVG (HEIGHT),2)AS 평균키
FROM PLAYER;

2.
---포지션 별 최대키,최소키,평균키를 출력( 포지션 별이란 소그룹의 조건을 제시했기 떄문에 GROUP BY 절을 사용한다.).
SELECT POSITION AS 포지션, COUNT(*) AS 인원수,COUNT (HEIGHT) AS 키대상 
,MAX(HEIGHT)AS 최대키, MIN(HEIGHT)AS 최소키 
,ROUND (AVG(HEIGHT),2)AS 평균키
FROM PLAYER
GROUP BY POSITION;

3.
---K리그 선수들의 포지션 별 평균키를 구하는데, 평균키가 180CM이상인 정보만 표시하라는 요구사항이 접수됐으므로
---WHERE 절과 GROUP BY 절을 사용해 SQL문장을 작성한다.

SELECT POSITION AS 포지션, ROUND(AVG(HEIGHT),2)AS평균키
FROM PLAYER
WHERE AVG(HEIGHT)>=180 
GROUP BY POSITION;

4.
--HAVING 조건절에는 GROUP BY 절에서 정의한 소그룹의 집계함수를 이용한 조건을 표시할 수 있으므로, HAVING절을 
---이용해 평균키가 180CM이상인 정보만 표시한다. 
SELECT POSITION AS 포지션, ROUND(AVG(HEIGHT),2)AS평균키
FROM PLAYER
GROUP BY POSITION
HAVING AVG(HEIGHT)>=180;

5.
---SQL 문장은 GROUP BY 절과 HAVING 절의 순서를 바꾸어서 수행한다. 
SELECT POSITION AS 포지션 , AVG(HEIGHT)AS 평균키 
FROM PLAYER 
HAVING AVG (HEIGHT)>=180
GROUP BY POSITION;

6.
---K리그의 선수들 중 삼성블루윙즈(K02)와 FC서울(K09)의 인원수는 얼마인가란 요구사항이 접수됐다. WHERE 절과 
---GROUP BY 절을 사용한 SQL과 GROUP BY 절과 HAVING 절을 사용한 SQL을 모두 작성한다.
SELECT TEAM_ID AS 팀ID, COUNT(*) AS 인원수 
FROM PLAYER
WHERE TEAM_ID IN('K09','K02')
GROUP BY TEAM_ID;

7.
---포지션별 평균키만 출력하는데, 최대키가 190CM이상인 선수를 갖고 있는 포지션의 정보만 출력한다. 
SELECT POSITION AS 포지션, ROUND(AVG(HEIGHT),2) AS 평균키 
FROM PLAYER 
GROUP BY POSITION
HAVING MAX (HEIGHT)>= 190;
